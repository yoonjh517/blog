### Spread operator

```jsx
let [title, setTitle] = useState([title1, title2, title3]);

function changeTitle() {
  var newArray = [...title];
  newArray[0] = "new title";
  setTitle(newArray);
}

<p>{title[0]}</p>

<button onClick={changeTitle}>CHANGE TITLE</button>;
```

### Conditional rendering

instead of if

```jsx
function App(){

  let [modal, setModal] = useState(false);

  return (
    // same code
    {
      modal === true
      ? <Modal />
      : null
    }

    <button onClick={()=>{setModal(true)}}>OPEN MODAL</button>
    <button onClick={()=>{setModal(!modal)}}>OPEN & CLOSE MODAL</button>
  )
}
```

### e.target.value

save input from client into the state

```jsx highlight={6}
function App() {
  let [input, setInput] = useState("");

  return (
    <div>
      <input
        onChange={(e) => {
          setInput(e.target.value);
        }}
      />
    </div>
  );
}
```

#### make another list item with input

```jsx
function App() {
  let [title, setTitle] = useState(["title 1", "title 2", "title 3"]);
  let [input, setInput] = useState("");

  return (
    <div>
      <input
        onChange={(e) => {
          setInput(e.target.value);
        }}
      />
      <button
        onClick={() => {
          let arrayCopy = [...title];
          arrayCopy.unshift(input);
          setTitle(arrayCopy);
        }}
      >
        SAVE
      </button>
    </div>
  );
}
```
